# ---------------------------------------------------------------------------

# Treated but stable only 3 weeks → excluded

    p := json.patch(_eligible_patient, [
        {"op": "replace", "path": "/patient/tumor/brain_metastases/present",                     "value": true},
        {"op": "replace", "path": "/patient/tumor/brain_metastases/treated",                     "value": true},
        {"op": "replace", "path": "/patient/tumor/brain_metastases/stable_months",               "value": 0.75},

        {"op": "replace", "path": "/patient/medications/lag3_antibody_prior",          "value": true}

    count(r) == 5
